\begin{frame}[fragile]
\frametitle{示例}
\begin{kotlincode}[basicstyle=\scriptsize\ttfamily,emph={[1]level}]
/**
 * 按照层级取出编码
 *
 * eg. "000a00".extract(2) -> "000a"
 */
fun String.extact(level: Int) = when (level) {
    0 -> this
    else -> this.substring(0, 2 * level)
}
\end{kotlincode}
\end{frame}
